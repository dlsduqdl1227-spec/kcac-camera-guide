  <script>
    (function() {
      var canvas = document.getElementById('canvas');
      var ctx = canvas.getContext('2d');
      var stage = document.getElementById('stage');
      var video = document.getElementById('video');
      var empty = document.getElementById('empty');
      var statusEl = document.getElementById('status');
      var fileInput = document.getElementById('fileInput');
      var captureBtn = document.getElementById('captureBtn');
      var cameraBtn = document.getElementById('cameraBtn');
      var uploadBtn = document.getElementById('uploadBtn');
      var fitBtn = document.getElementById('fitBtn');
      var resetBtn = document.getElementById('resetBtn');
      var saveBtn = document.getElementById('saveBtn');
      var downloadBtn = document.getElementById('downloadBtn');
      var guideSelect = document.getElementById('guideSelect');
      var radiusRange = document.getElementById('radiusRange');
      var opacityRange = document.getElementById('opacityRange');
      var guideLabel = document.getElementById('guideLabel');
      var radiusLabel = document.getElementById('radiusLabel');
      var opacityLabel = document.getElementById('opacityLabel');
      var thumbs = document.getElementById('thumbs');
      var modal = document.getElementById('modal');
      var modalImg = document.getElementById('modalImg');
      var modalClose = document.getElementById('modalClose');

      var state = {
        image: null,
        stream: null,
        fit: 'contain',
        guide: 'circle',
        opacity: 0.85,
        circle: { x: 0.5, y: 0.5, r: 0.32 },
        dragging: false,
        snapshots: []
      };

      function setStatus(text) {
        statusEl.textContent = text;
      }

      function resizeCanvas() {
        var rect = stage.getBoundingClientRect();
        var dpr = window.devicePixelRatio || 1;
        canvas.width = Math.max(1, Math.round(rect.width * dpr));
        canvas.height = Math.max(1, Math.round(rect.height * dpr));
        ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
        draw();
      }

      function imageRect(img) {
        var w = canvas.clientWidth;
        var h = canvas.clientHeight;
        var iw = img.videoWidth || img.naturalWidth || img.width || 1;
        var ih = img.videoHeight || img.naturalHeight || img.height || 1;
        var scale = state.fit === 'cover' ? Math.max(w / iw, h / ih) : Math.min(w / iw, h / ih);
        var dw = iw * scale;
        var dh = ih * scale;
        return { x: (w - dw) / 2, y: (h - dh) / 2, w: dw, h: dh };
      }

      function drawGuide(rect) {
        var alpha = state.opacity;
        var cx = rect.x + rect.w * state.circle.x;
        var cy = rect.y + rect.h * state.circle.y;
        var r = Math.min(rect.w, rect.h) * state.circle.r;
        ctx.save();
        ctx.globalAlpha = alpha;
        ctx.strokeStyle = '#5ee0b7';
        ctx.fillStyle = '#5ee0b7';
        ctx.lineWidth = 2;
        ctx.setLineDash([8, 8]);

        if (state.guide === 'circle' || state.guide === 'concentric') {
          ctx.beginPath();
          ctx.arc(cx, cy, r, 0, Math.PI * 2);
          ctx.stroke();
        }
        if (state.guide === 'concentric') {
          ctx.beginPath();
          ctx.arc(cx, cy, r * 0.66, 0, Math.PI * 2);
          ctx.stroke();
          ctx.beginPath();
          ctx.arc(cx, cy, r * 0.33, 0, Math.PI * 2);
          ctx.stroke();
        }
        if (state.guide === 'cross' || state.guide === 'circle' || state.guide === 'concentric') {
          ctx.beginPath();
          ctx.moveTo(cx - r, cy);
          ctx.lineTo(cx + r, cy);
          ctx.moveTo(cx, cy - r);
          ctx.lineTo(cx, cy + r);
          ctx.stroke();
        }
        if (state.guide === 'thirds') {
          ctx.setLineDash([6, 6]);
          ctx.beginPath();
          for (var i = 1; i <= 2; i++) {
            ctx.moveTo(rect.x + rect.w * i / 3, rect.y);
            ctx.lineTo(rect.x + rect.w * i / 3, rect.y + rect.h);
            ctx.moveTo(rect.x, rect.y + rect.h * i / 3);
            ctx.lineTo(rect.x + rect.w, rect.y + rect.h * i / 3);
          }
          ctx.stroke();
        }
        if (state.guide === 'diagonal') {
          ctx.beginPath();
          ctx.moveTo(rect.x, rect.y);
          ctx.lineTo(rect.x + rect.w, rect.y + rect.h);
          ctx.moveTo(rect.x + rect.w, rect.y);
          ctx.lineTo(rect.x, rect.y + rect.h);
          ctx.stroke();
        }
        ctx.setLineDash([]);
        ctx.beginPath();
        ctx.arc(cx, cy, 5, 0, Math.PI * 2);
        ctx.fill();
        ctx.restore();
      }

      function draw() {
        ctx.clearRect(0, 0, canvas.clientWidth, canvas.clientHeight);
        var source = state.stream ? video : state.image;
        empty.style.display = source ? 'none' : 'flex';
        if (!source) return;
        var rect = imageRect(source);
        ctx.drawImage(source, rect.x, rect.y, rect.w, rect.h);
        drawGuide(rect);
        if (state.stream) requestAnimationFrame(draw);
      }

      function setImageFromDataUrl(url) {
        var img = new Image();
        img.onload = function() {
          state.image = img;
          stopCamera(false);
          setStatus('사진 불러오기 완료');
          draw();
        };
        img.onerror = function() { setStatus('사진을 불러오지 못했습니다.'); };
        img.src = url;
      }

      function stopCamera(updateStatus) {
        if (state.stream) {
          state.stream.getTracks().forEach(function(track) { track.stop(); });
        }
        state.stream = null;
        video.srcObject = null;
        captureBtn.disabled = true;
        cameraBtn.textContent = '카메라 시작';
        if (updateStatus) setStatus('카메라 중지');
      }

      function exportDataUrl() {
        var out = document.createElement('canvas');
        out.width = canvas.width;
        out.height = canvas.height;
        var octx = out.getContext('2d');
        octx.drawImage(canvas, 0, 0);
        return out.toDataURL('image/png');
      }

      function addSnapshot(url) {
        state.snapshots.unshift(url);
        state.snapshots = state.snapshots.slice(0, 12);
        thumbs.innerHTML = '';
        state.snapshots.forEach(function(src) {
          var btn = document.createElement('button');
          btn.className = 'thumb';
          btn.innerHTML = '<img alt="스냅샷" src="' + src + '">';
          btn.onclick = function() {
            modalImg.src = src;
            modal.classList.add('on');
          };
          thumbs.appendChild(btn);
        });
      }

      function pointerToCircle(evt) {
        var rect = imageRect(state.stream ? video : state.image);
        var x = (evt.clientX - rect.x) / rect.w;
        var y = (evt.clientY - rect.y) / rect.h;
        state.circle.x = Math.max(0, Math.min(1, x));
        state.circle.y = Math.max(0, Math.min(1, y));
        draw();
      }

      uploadBtn.onclick = function() { fileInput.click(); };
      fileInput.onchange = function() {
        var file = fileInput.files && fileInput.files[0];
        if (!file) return;
        var reader = new FileReader();
        reader.onload = function(e) { setImageFromDataUrl(e.target.result); };
        reader.readAsDataURL(file);
      };

      cameraBtn.onclick = function() {
        if (state.stream) { stopCamera(true); draw(); return; }
        if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) {
          setStatus('이 브라우저는 카메라를 지원하지 않습니다.');
          return;
        }
        navigator.mediaDevices.getUserMedia({ video: { facingMode: 'environment' }, audio: false })
          .then(function(stream) {
            state.stream = stream;
            state.image = null;
            video.srcObject = stream;
            cameraBtn.textContent = '카메라 중지';
            captureBtn.disabled = false;
            setStatus('카메라 실행 중');
            draw();
          })
          .catch(function(err) {
            setStatus('카메라 권한을 확인해주세요: ' + (err && err.message ? err.message : err));
          });
      };

      captureBtn.onclick = function() {
        if (!state.stream) return;
        var url = exportDataUrl();
        addSnapshot(url);
        setImageFromDataUrl(url);
      };

      saveBtn.onclick = function() {
        if (!state.image && !state.stream) { setStatus('저장할 이미지가 없습니다.'); return; }
        addSnapshot(exportDataUrl());
        setStatus('스냅샷 저장 완료');
      };

      downloadBtn.onclick = function() {
        if (!state.image && !state.stream) { setStatus('다운로드할 이미지가 없습니다.'); return; }
        var a = document.createElement('a');
        a.href = exportDataUrl();
        a.download = 'KCAC_guide_' + new Date().toISOString().replace(/[:.]/g, '-') + '.png';
        document.body.appendChild(a);
        a.click();
        a.remove();
      };

      fitBtn.onclick = function() {
        state.fit = state.fit === 'contain' ? 'cover' : 'contain';
        fitBtn.textContent = state.fit === 'contain' ? '맞춤 전환' : '채움 전환';
        draw();
      };

      resetBtn.onclick = function() {
        state.circle = { x: 0.5, y: 0.5, r: 0.32 };
        radiusRange.value = '32';
        radiusLabel.textContent = '32%';
        draw();
      };

      guideSelect.onchange = function() {
        state.guide = guideSelect.value;
        guideLabel.textContent = guideSelect.options[guideSelect.selectedIndex].text;
        draw();
      };
      radiusRange.oninput = function() {
        state.circle.r = Number(radiusRange.value) / 100;
        radiusLabel.textContent = radiusRange.value + '%';
        draw();
      };
      opacityRange.oninput = function() {
        state.opacity = Number(opacityRange.value) / 100;
        opacityLabel.textContent = opacityRange.value + '%';
        draw();
      };

      canvas.addEventListener('pointerdown', function(evt) {
        if (!state.image && !state.stream) return;
        state.dragging = true;
        canvas.setPointerCapture(evt.pointerId);
        pointerToCircle(evt);
      });
      canvas.addEventListener('pointermove', function(evt) {
        if (!state.dragging) return;
        pointerToCircle(evt);
      });
      canvas.addEventListener('pointerup', function(evt) {
        state.dragging = false;
        try { canvas.releasePointerCapture(evt.pointerId); } catch (e) {}
      });
      canvas.addEventListener('pointercancel', function() { state.dragging = false; });

      modalClose.onclick = function() { modal.classList.remove('on'); };
      modal.onclick = function(evt) {
        if (evt.target === modal) modal.classList.remove('on');
      };
      window.addEventListener('resize', resizeCanvas);
      resizeCanvas();
    })();
  </script>
</body>
